def crear_tabla():
    try:
        conn = sql.connect(**DB_CONFIG)
        cursor = conn.cursor()
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS proyectos (
                id INT AUTO_INCREMENT PRIMARY KEY,
                titulo VARCHAR(150) NOT NULL UNIQUE,
                resumen TEXT NOT NULL,
                facultad VARCHAR(255) NOT NULL,
                programa VARCHAR(255) NOT NULL,
                tipo TEXT NOT NULL,
                palabras_clave TEXT NOT NULL,
                estado VARCHAR(35) NOT NULL,
                estudiantes TEXT NOT NULL,
                docente TEXT NOT NULL,
                registrado VARCHAR(80) NOT NULL,
                fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        conn.commit()
        conn.close()
        print("Tabla creada correctamente.")
    except sql.Error as e:
        print("Error al crear la tabla:", e)

        
def insertar_proyectos(proyectos):
    try:
        conn = sql.connect(**DB_CONFIG)
        cursor = conn.cursor()
        
        for proyecto in proyectos:
            if len(proyecto) != 10:
                print(f"Proyecto con número de parámetros incorrecto: {proyecto}")
                continue
            
            try:
                cursor.execute("""
                    INSERT INTO proyectos (titulo, resumen, facultad, programa, tipo, palabras_clave, estado, estudiantes, docente, registrado)
                    VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s)
                """, proyecto)
                conn.commit()
            except sql.Error as e:
                print(f"Error al insertar el proyecto {proyecto[0]}: {e}")
                conn.rollback()  # Revertir cambios si hay un error
        
        conn.close()
        print("Datos insertados correctamente.")
    except sql.Error as e:
        print("Error al conectar a la base de datos:", e)

# Crear la tabla
crear_tabla()

# Insertar los proyectos
insertar_proyectos(proyectos)
